\begin{verbatim}
if [ $# -ne 1 ]; then
    echo "Usage: $0 {1|2|3|4}"
    echo "1. Addition"
    echo "2. Subtraction"
    echo "3. Multiplication"
    echo "4. Division"
    exit 1
fi
read -p "Enter first number: " a
read -p "Enter second number: " b
case $1 in
    1)
        echo "Result: $((a + b))"
        ;;
    2)
        echo "Result: $((a - b))"
        ;;
    3)
        echo "Result: $((a * b))"
        ;;
    4)
        if [ "$b" -eq 0 ]; then
            echo "Error: Division by zero!"
        else
            echo "Result: $((a / b))"
        fi
        ;;
    *)
        echo "Invalid choice. Use: 1 (Add), 2 (Sub), 
        3 (Mul), 4 (Div)"
        ;;
esac
\end{verbatim}
