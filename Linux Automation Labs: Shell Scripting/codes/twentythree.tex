\begin{verbatim}
rows=4
for (( i=1; i<=rows; i++ ))
do
    for (( s=rows-i; s>0; s-- ))
    do
        echo -n " "
    done
    for (( j=1; j<=2*i-1; j++ ))
    do
        echo -n "*"
    done
    echo
done
for (( i=rows-1; i>=1; i-- ))
do
    for (( s=rows-i; s>0; s-- ))
    do
        echo -n " "
    done
    for (( j=1; j<=2*i-1; j++ ))
    do
        echo -n "*"
    done
    echo
done
\end{verbatim}
