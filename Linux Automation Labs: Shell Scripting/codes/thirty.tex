\begin{verbatim}
echo "Enter sorted array elements (space-separated):"
read -a arr
read -p "Enter the number to search: " key
low=0
high=$((${#arr[@]} - 1))
found=0
while [ $low -le $high ]
do
    mid=$(( (low + high) / 2 ))
    if [ "${arr[$mid]}" -eq "$key" ]; then
        echo "Number $key found at position $((mid+1))."
        found=1
        break
    elif [ "${arr[$mid]}" -lt "$key" ]; then
        low=$((mid + 1))
    else
        high=$((mid - 1))
    fi
done
if [ $found -eq 0 ]; then
    echo "Number $key not found in the array."
fi
\end{verbatim}
