\begin{enumerate}
    \item \textbf{Start}
    \item Prompt the user to enter numbers separated by space
    \item Read the input numbers into an array \texttt{arr}
    \item Determine the number of elements \texttt{n} in the array
    \item Perform Bubble Sort:
    \begin{itemize}
        \item For each index \texttt{i} from 0 to \texttt{n - 2}
        \begin{itemize}
            \item For each index \texttt{j} from 0 to \texttt{n - i - 2}
            \begin{itemize}
                \item If \texttt{arr[j]} > \texttt{arr[j+1]}, then:
                \begin{itemize}
                    \item Swap \texttt{arr[j]} and \texttt{arr[j+1]}
                \end{itemize}
            \end{itemize}
        \end{itemize}
    \end{itemize}
    \item Display the sorted array
    \item \textbf{End}
\end{enumerate}
