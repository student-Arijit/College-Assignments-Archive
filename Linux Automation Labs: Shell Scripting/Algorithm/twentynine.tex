\begin{enumerate}
    \item \textbf{Start}
    \item Prompt the user to enter array elements (space-separated) and store them in array \texttt{arr}
    \item Prompt the user to enter the number to search and store it in variable \texttt{key}
    \item Initialize a flag variable \texttt{found} to 0
    \item Loop through the array from index 0 to length - 1
    \begin{itemize}
        \item If \texttt{arr[i] == key}, then:
        \begin{itemize}
            \item Print the position of the key (i+1)
            \item Set \texttt{found = 1}
            \item Exit the loop
        \end{itemize}
    \end{itemize}
    \item After the loop, if \texttt{found == 0}, print ``Number not found in the array''
    \item \textbf{End}
\end{enumerate}
