\begin{enumerate}
    \item Start
    \item Read the filename from the command-line argument
    \item If the number of arguments is not equal to 1:
    \begin{enumerate}
        \item Display usage message and exit
    \end{enumerate}
    \item Check if the file exists:
    \begin{enumerate}
        \item If not, display an error message and exit
    \end{enumerate}
    \item Sort the contents of the file
    \item Remove duplicate lines using a uniqueness filter
    \item Save the result in a file named \texttt{unique\_lines.txt}
    \item Display a message confirming the output
    \item End
\end{enumerate}
