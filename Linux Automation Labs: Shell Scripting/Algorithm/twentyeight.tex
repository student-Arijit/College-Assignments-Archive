\begin{enumerate}
    \item \textbf{Start}
    \item Retrieve the current hour using \texttt{date +\%H} and store it in variable \texttt{hour}
    \item Check the value of \texttt{hour}:
    \begin{itemize}
        \item If $5 \leq \texttt{hour} < 12$, display ``Good Morning!''
        \item Else if $12 \leq \texttt{hour} < 17$, display ``Good Afternoon!''
        \item Else if $17 \leq \texttt{hour} < 21$, display ``Good Evening!''
        \item Else, display ``Good Night!''
    \end{itemize}
    \item \textbf{End}
\end{enumerate}
