\begin{enumerate}
    \item \textbf{Start}
    \item Prompt the user to enter a word and store it in variable \texttt{word}
    \item Calculate the length of \texttt{word} and store it in \texttt{length}
    \item If \texttt{length < 4}, then:
    \begin{itemize}
        \item Display ``The word must be at least 4 characters long.''
        \item Exit the program
    \end{itemize}
    \item Reverse the word and store it in \texttt{reverse}
    \item Compare \texttt{word} and \texttt{reverse}
    \begin{itemize}
        \item If they are equal, display ``The word is a palindrome.''
        \item Else, display ``The word is not a palindrome.''
    \end{itemize}
    \item \textbf{End}
\end{enumerate}
