\begin{enumerate}
    \item Start
    \item Read two command-line arguments: \texttt{input\_file} and \texttt{output\_file}
    \item If the number of arguments is not equal to 2:
    \begin{enumerate}
        \item Display usage message and exit
    \end{enumerate}
    \item Check if \texttt{input\_file} exists:
    \begin{enumerate}
        \item If not, display an error message and exit
    \end{enumerate}
    \item Read the contents of \texttt{input\_file}
    \item Remove all vowels (both uppercase and lowercase) from the content
    \item Write the result to \texttt{output\_file}
    \item Display confirmation message
    \item End
\end{enumerate}
