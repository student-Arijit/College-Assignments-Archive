\begin{enumerate}
    \item Start
    \item Check if the number of command-line arguments is not equal to 1
    \begin{itemize}
        \item If true, display usage message and exit
    \end{itemize}
    \item Read two numbers: $a$ and $b$ from user
    \item Based on the command-line argument, perform:
    \begin{itemize}
        \item Case 1: Addition
        \begin{itemize}
            \item Compute and display $a + b$
        \end{itemize}
        \item Case 2: Subtraction
        \begin{itemize}
            \item Compute and display $a - b$
        \end{itemize}
        \item Case 3: Multiplication
        \begin{itemize}
            \item Compute and display $a \times b$
        \end{itemize}
        \item Case 4: Division
        \begin{itemize}
            \item If $b = 0$, display "Error: Division by zero"
            \item Else compute and display $a \div b$
        \end{itemize}
        \item Default: Display "Invalid choice"
    \end{itemize}
    \item End
\end{enumerate}
