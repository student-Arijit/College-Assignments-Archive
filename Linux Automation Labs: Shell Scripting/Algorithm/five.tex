 \begin{enumerate}
    \item Start
    \item Read the filename from the command-line argument
    \item Check if the filename is provided
    \begin{enumerate}
        \item If not, display usage message and exit
    \end{enumerate}
    \item Check if the file exists
    \begin{enumerate}
        \item If not, display error message and exit
    \end{enumerate}
    \item Use a pattern-matching command to find lines starting with a vowel (A, E, I, O, U or a, e, i, o, u)
    \item Display the matching lines
    \item End
\end{enumerate}
