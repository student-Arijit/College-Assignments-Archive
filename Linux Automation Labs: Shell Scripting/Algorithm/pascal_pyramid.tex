        \begin{itemize}
            \item If \texttt{$j == 0$} or \texttt{$i == 0$}, set \texttt{$coef = 1$}.
            \item Else, calculate using recurrence:
            \[
            C(i, j) = C(i, j-1) \times \frac{(i - j + 1)}{j}
            \]
            \item Print the coefficient with formatting.
        \end{itemize}
    \end{itemize}
    \item Move to the next line after each row.
\end{enumerate}
