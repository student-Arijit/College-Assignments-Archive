\begin{enumerate}
    \item Start
    \item Read command-line arguments: filename, $m$, and $n$
    \item Check if all three arguments are provided
    \begin{enumerate}
        \item If not, display usage message and exit
    \end{enumerate}
    \item Check if the file exists
    \begin{enumerate}
        \item If not, display error message and exit
    \end{enumerate}
    \item Use a text-processing command to print lines from $m$ to $n$
    \item Display the output
    \item End
\end{enumerate}
