\begin{enumerate}
    \item Start
    \item Read the folder name from the command-line argument
    \item If no argument is given:
    \begin{enumerate}
        \item Display usage message and exit
    \end{enumerate}
    \item Check if the argument is a valid directory
    \begin{enumerate}
        \item If not, display error and exit
    \end{enumerate}
    \item Create or clear an output file named \texttt{empty\_folders.txt}
    \item Traverse the folder recursively
    \item For each subfolder:
    \begin{enumerate}
        \item If it is empty, add its name to \texttt{empty\_folders.txt}
    \end{enumerate}
    \item Display a message with the output file name
    \item End
\end{enumerate}
