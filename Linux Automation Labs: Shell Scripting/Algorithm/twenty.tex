\begin{enumerate}
    \item \textbf{Start}
    \item Check if the number of command-line arguments is not equal to 1
    \begin{itemize}
        \item If true, display usage message: \texttt{Usage: \$0 number}
        \item Exit the script
    \end{itemize}
    \item Assign the first command-line argument to variable \texttt{num}
    \item Reverse the contents of \texttt{num} using \texttt{rev} command and store in variable \texttt{rev}
    \item Compare \texttt{num} and \texttt{rev}
    \begin{itemize}
        \item If equal, print: \texttt{<num> is a palindrome}
        \item Else, print: \texttt{<num> is not a palindrome}
    \end{itemize}
    \item \textbf{End}
\end{enumerate}
