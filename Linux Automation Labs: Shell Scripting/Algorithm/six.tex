\begin{enumerate}
    \item Start
    \item Read all command-line arguments
    \item If no arguments are given:
    \begin{enumerate}
        \item Display usage message and exit
    \end{enumerate}
    \item For each argument:
    \begin{enumerate}
        \item If it is a regular file:
        \begin{enumerate}
            \item Display that it is a file
            \item Count and display the number of lines in it
        \end{enumerate}
        \item Else if it is a directory:
        \begin{enumerate}
            \item Display that it is a directory
        \end{enumerate}
        \item Else:
        \begin{enumerate}
            \item Display that it is neither a file nor a directory
        \end{enumerate}
    \end{enumerate}
    \item End
\end{enumerate}
