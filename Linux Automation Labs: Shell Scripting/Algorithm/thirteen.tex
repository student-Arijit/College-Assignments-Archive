\begin{enumerate}
    \item Start
    \item Read the filename from the command-line argument
    \item If the number of arguments is not 1:
    \begin{enumerate}
        \item Display usage message and exit
    \end{enumerate}
    \item Check if the file exists:
    \begin{enumerate}
        \item If not, display error message and exit
    \end{enumerate}
    \item Process the file:
    \begin{enumerate}
        \item Replace spaces and punctuation with newlines to extract words
        \item Convert all characters to lowercase
        \item Remove empty lines
        \item Sort the words
        \item Count the frequency of each unique word
        \item Sort the result in descending order of frequency
    \end{enumerate}
    \item Display the result (word frequencies)
    \item End
\end{enumerate}
