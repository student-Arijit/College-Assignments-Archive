\begin{enumerate}
    \item Start
    \item Read sorting order from command-line argument
    \begin{enumerate}
        \item If 1, sort in ascending order
        \item If 2, sort in descending order
    \end{enumerate}
    \item Prompt the user to enter the number of elements
    \item Read the array elements
    \item Perform bubble sort:
    \begin{enumerate}
        \item For each element from $0$ to $n - 2$:
        \begin{enumerate}
            \item Compare adjacent elements
            \item If out of order (based on selected mode), swap them
        \end{enumerate}
    \end{enumerate}
    \item Display the sorted array
    \item End
\end{enumerate}
