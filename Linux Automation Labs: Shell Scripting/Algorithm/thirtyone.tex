\begin{enumerate}
    \item \textbf{Start}
    \item Prompt the user to enter a non-negative integer and store it in variable \texttt{num}
    \item If the input is not a non-negative integer:
    \begin{itemize}
        \item Display error message
        \item Exit the program
    \end{itemize}
    \item Initialize \texttt{fact = 1}
    \item For \texttt{i} from 1 to \texttt{num}, repeat:
    \begin{itemize}
        \item Multiply \texttt{fact} by \texttt{i}
        \item Update \texttt{fact}
    \end{itemize}
    \item Display the factorial result \texttt{fact}
    \item \textbf{End}
\end{enumerate}
