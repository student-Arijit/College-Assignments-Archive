\begin{enumerate}
    \item Start
    \item Read the range (number of terms) from command-line argument
    \item If no argument is given:
    \begin{enumerate}
        \item Display usage message and exit
    \end{enumerate}
    \item Initialize two variables: $a = 0$, $b = 1$
    \item Repeat for the given range:
    \begin{enumerate}
        \item Print $a$
        \item Calculate next term as $fn = a + b$
        \item Set $a = b$
        \item Set $b = fn$
    \end{enumerate}
    \item End
\end{enumerate}
