\begin{enumerate}
    \item Start
    \item Read two numbers: $a$ and $b$
    \item Remove any negative sign from both $a$ and $b$
    \item While $b \ne 0$, repeat:
    \begin{enumerate}
        \item Set \texttt{temp} = $b$
        \item Set $b = a \mod b$
        \item Set $a = \texttt{temp}$
    \end{enumerate}
    \item When loop ends, $a$ is the GCD
    \item Print "GCD is: $a$"
    \item End
\end{enumerate}
