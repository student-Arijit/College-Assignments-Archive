    \item \textbf{Start}
    \item Prompt the user to enter a string and store it in variable \texttt{str}
    \item Reverse the string using the \texttt{rev} command and store it in \texttt{rev\_str}
    \item Compare \texttt{str} with \texttt{rev\_str}
    \begin{itemize}
        \item If both are equal, display: "\texttt{str} is a palindrome"
        \item Else, display: "\texttt{str} is not a palindrome"
    \end{itemize}
    \item \textbf{End}
\end{enumerate}
