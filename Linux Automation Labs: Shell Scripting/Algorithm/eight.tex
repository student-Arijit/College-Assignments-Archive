\begin{enumerate}
    \item Start
    \item Read the filename from the command-line argument
    \item If no filename is provided:
    \begin{enumerate}
        \item Display usage message and exit
    \end{enumerate}
    \item Check if the file exists
    \begin{enumerate}
        \item If not, display error message and exit
    \end{enumerate}
    \item Count the number of lines in the file
    \item Count the number of words in the file
    \item Count the number of characters in the file
    \item Display the counts in the format: lines words characters filename
    \item End
\end{enumerate}
