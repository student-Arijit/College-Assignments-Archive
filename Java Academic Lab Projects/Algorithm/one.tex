\begin{algorithm}[H]
\caption{Half Sort Array (First Half Ascending, Second Half Descending)}
\begin{algorithmic}[1]
\STATE START
\STATE INPUT THE SIZE OF THE ARRAY $N$
\STATE DECLARE AN INTEGER ARRAY $ARR[N]$
\FOR{$I = 0$ TO $N - 1$}
    \STATE INPUT $ARR[I]$
\ENDFOR
\STATE CALCULATE MIDPOINT $MID = N / 2$
\STATE SORT THE FIRST HALF OF THE ARRAY $ARR[0 \text{ TO } MID - 1]$ IN ASCENDING ORDER
\STATE CREATE A TEMPORARY INTEGER ARRAY $SECONDHALF[N - MID]$
\FOR{$I = MID$ TO $N - 1$}
    \STATE $SECONDHALF[I - MID] \gets ARR[I]$
\ENDFOR
\STATE SORT $SECONDHALF$ IN DESCENDING ORDER
\FOR{$I = MID$ TO $N - 1$}
    \STATE $ARR[I] \gets SECONDHALF[I - MID]$
\ENDFOR
\STATE OUTPUT THE MODIFIED ARRAY $ARR$
\STATE STOP
\end{algorithmic}
\end{algorithm}
