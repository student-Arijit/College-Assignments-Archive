\begin{algorithm}[H]
\caption{Class \texttt{MyPoint} Functionality}
\begin{algorithmic}[1]
\STATE \textbf{Class} MyPoint
\STATE Declare private integer variables \texttt{x}, \texttt{y}

\STATE \textbf{Constructor} MyPoint()
\STATE \quad Set \texttt{x = 0}, \texttt{y = 0}

\STATE \textbf{Constructor} MyPoint(x, y)
\STATE \quad Set \texttt{this.x = x}, \texttt{this.y = y}

\STATE \textbf{Method} setXY(x, y)
\STATE \quad Set \texttt{this.x = x}, \texttt{this.y = y}

\STATE \textbf{Method} getXY()
\STATE \quad Return array \texttt{\{x, y\}}

\STATE \textbf{Method} toString()
\STATE \quad Return string \texttt{"(x, y)"}

\STATE \textbf{Method} distance(x, y)
\STATE \quad Compute \texttt{dx = this.x - x}
\STATE \quad Compute \texttt{dy = this.y - y}
\STATE \quad Return \texttt{sqrt(dx * dx + dy * dy)}
\end{algorithmic}
\end{algorithm}
