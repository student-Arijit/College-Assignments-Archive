\begin{algorithm}[H]
\caption{Remove Consonants From a String}
\begin{algorithmic}[1]
\STATE START
\STATE INPUT A STRING FROM THE USER AND STORE IT IN $\textbf{INPUT}$
\STATE INITIALIZE AN EMPTY STRING BUILDER $\textbf{RESULT}$
\FOR{EACH CHARACTER $\textbf{CH}$ IN $\textbf{INPUT}$}
    \IF{$CH$ IS A VOWEL OR NOT A LETTER}
        \STATE APPEND $\textbf{CH}$ TO $\textbf{RESULT}$
    \ENDIF
\ENDFOR
\STATE DISPLAY $\textbf{RESULT}$ AS THE STRING AFTER REMOVING CONSONANTS
\STATE STOP
\end{algorithmic}
\end{algorithm}
