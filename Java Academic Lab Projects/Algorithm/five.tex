\begin{algorithm}[H]
\caption{Class \texttt{Person}, \texttt{Student}, and \texttt{Staff} Structure}
\begin{algorithmic}[1]
\STATE \textbf{Class} Person
\STATE \quad Declare private variables: \texttt{name}, \texttt{address}
\STATE \quad \textbf{Constructor}(name, address): Initialize fields
\STATE \quad \textbf{Method} setPerson(name, address): Set new values
\STATE \quad \textbf{Method} toString(): Return "Person[name=..., address=...]"

\STATE \textbf{Class} Student \textbf{extends} Person
\STATE \quad Declare private variables: \texttt{program}, \texttt{year}, \texttt{fees}
\STATE \quad \textbf{Constructor}(name, address, program, year, fees): 
\STATE \quad \quad Call \texttt{super(name, address)}, set remaining fields
\STATE \quad \textbf{Method} setStudent(...): Update inherited and own fields
\STATE \quad \textbf{Override toString()}: Return full student details

\STATE \textbf{Class} Staff \textbf{extends} Person
\STATE \quad Declare private variables: \texttt{school}, \texttt{pay}
\STATE \quad \textbf{Constructor}(name, address, school, pay): 
\STATE \quad \quad Call \texttt{super(name, address)}, set remaining fields
\STATE \quad \textbf{Method} setStaff(...): Update inherited and own fields
\STATE \quad \textbf{Override toString()}: Return full staff details
\end{algorithmic}
\end{algorithm}
