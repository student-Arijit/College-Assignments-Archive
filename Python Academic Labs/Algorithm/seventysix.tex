\begin{algorithm}
\caption{Find Eigenvalues of a 3×3 Matrix}
\begin{algorithmic}[1]

\STATE Import NumPy library

\STATE Define a 3×3 matrix:
\STATE $A \gets \begin{bmatrix}
4 & 2 & 1 \\
0 & 3 & -1 \\
0 & 0 & 2
\end{bmatrix}$

\STATE Compute eigenvalues of matrix $A$:
\STATE $eigenvalues \gets eigvals(A)$

\STATE Display matrix $A$

\STATE Display eigenvalues of $A$

\end{algorithmic}
\end{algorithm}
