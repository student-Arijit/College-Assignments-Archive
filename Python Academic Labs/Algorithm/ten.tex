\begin{algorithm}
\caption{Display Number Pattern}
\begin{algorithmic}[1]

\FOR{$i \gets 1$ to $5$}

    \FOR{$j \gets 1$ to $i$}
        \STATE Output $i$
    \ENDFOR

    \STATE Output space

    \FOR{$j \gets i$ down to $1$}
        \STATE Output $j$
    \ENDFOR

    \STATE Move to next line

\ENDFOR

\end{algorithmic}
\end{algorithm}
