\begin{algorithm}
\caption{Search an Item in a Tuple}
\begin{algorithmic}[1]

\STATE Create a tuple:
\STATE $data \gets (10,\ 20,\ 30,\ 40,\ 50)$

\STATE Input item to search:
\STATE $item \gets$ user input

\IF{$item$ exists in $data$}
    \STATE Output ``Item exists in the tuple.''
\ELSE
    \STATE Output ``Item does not exist in the tuple.''
\ENDIF

\end{algorithmic}
\end{algorithm}
