\begin{algorithm}
\caption{Compute Speed of Plane}
\begin{algorithmic}[1]
\REQUIRE Distance $d$, Time $t$
\ENSURE Speed in m/s and km/h

\STATE Input $d \gets 395000$
\STATE Input $t \gets 9000$

\STATE Compute speed:
\STATE $speed \gets \frac{d}{t}$

\STATE Compute speed in km/h:
\STATE $speed\_kmh \gets speed \times 3.6$

\STATE Output ``Speed: '' $speed$ `` m/s''
\STATE Output ``Speed: '' $speed\_kmh$ `` km/h''

\end{algorithmic}
\end{algorithm}
