\begin{algorithm}
\caption{Display Slices of an Alphabet List}
\begin{algorithmic}[1]

\STATE Create list of letters:
\STATE $letters \gets [a,b,c,d,e,f,g,h]$

\STATE Extract first three letters:
\STATE $first\_three \gets letters[0:3]$

\STATE Extract three letters from the middle:
\STATE $middle\_three \gets letters[2:5]$

\STATE Extract letters from index 4 to the end:
\STATE $remaining \gets letters[4:]$

\STATE Output ``First three letters: '' $first\_three$

\STATE Output ``Three letters from the middle: '' $middle\_three$

\STATE Output ``Letters from index 4 to end: '' $remaining$

\end{algorithmic}
\end{algorithm}
