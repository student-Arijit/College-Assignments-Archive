\begin{algorithm}
\caption{Calculate Acceleration of a Car}
\begin{algorithmic}[1]

\STATE Set initial velocity:
\STATE $v_{initial} \gets 0$

\STATE Set final velocity:
\STATE $v_{final} \gets 10$

\STATE Set time taken:
\STATE $time \gets 20$

\STATE Compute acceleration:
\STATE $acceleration \gets \dfrac{v_{final}-v_{initial}}{time}$

\STATE Output ``Acceleration: '' $acceleration$ `` m/s$^2$''

\end{algorithmic}
\end{algorithm}
