\begin{algorithm}
\caption{Addition of Two Polynomials Represented Using Coefficient Lists}
\begin{algorithmic}[1]

\STATE Input highest degree of polynomial:
\STATE $deg \gets$ user input

\STATE Input coefficients of first polynomial (highest to lowest degree):
\STATE $P_1 \gets$ list of coefficients

\STATE Input coefficients of second polynomial:
\STATE $P_2 \gets$ list of coefficients

\STATE Pad the shorter list with leading zeros to match lengths

\STATE Add corresponding coefficients:
\STATE $Result[i] \gets P_1[i] + P_2[i]$

\STATE Create polynomial objects for $P_1$, $P_2$, and $Result$

\STATE Display first polynomial in readable form

\STATE Display second polynomial in readable form

\STATE Display sum of polynomials in readable form

\end{algorithmic}
\end{algorithm}
