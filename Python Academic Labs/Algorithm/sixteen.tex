\begin{algorithm}
\caption{Sort a String Lexicographically}
\begin{algorithmic}[1]

\STATE Input string:
\STATE $text \gets$ user input

\STATE Convert string into a list of characters:
\STATE $chars \gets$ list of characters in $text$

\STATE Sort $chars$ in ascending order

\STATE Join all characters to form a string:
\STATE $sorted\_string \gets$ concatenate elements of $chars$

\STATE Output ``Lexicographically sorted string: '' $sorted\_string$

\end{algorithmic}
\end{algorithm}
