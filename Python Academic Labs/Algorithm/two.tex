\begin{algorithm}
\caption{Convert Kilogram to Pound Using Dataclass}
\begin{algorithmic}[1]

\STATE Start
\STATE Import dataclass from dataclasses module
\STATE Define class Weight with attribute kg (float)
\STATE Define property method pound
\STATE \hspace{0.5cm} Return kg $\times$ 2.20462
\STATE Read weight in kg from user
\STATE Convert input to integer
\STATE Create object w of class Weight using kg
\STATE Print w.pound
\STATE Stop

\end{algorithmic}
\end{algorithm}
