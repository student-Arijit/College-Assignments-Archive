\begin{algorithm}
\caption{Count the Occurrence of Each Letter in a File}
\begin{algorithmic}[1]

\STATE Input file name:
\STATE $filename \gets$ user input

\STATE Initialize an empty dictionary:
\STATE $freq \gets \{\}$

\STATE Open the file in read mode

\STATE Read all text from the file:
\STATE $text \gets$ contents of file

\FOR{each character $ch$ in $text$}
    \IF{$ch$ is an alphabet}
        \STATE Convert $ch$ to lowercase
        \IF{$ch$ exists in $freq$}
            \STATE $freq[ch] \gets freq[ch] + 1$
        \ELSE
            \STATE $freq[ch] \gets 1$
        \ENDIF
    \ENDIF
\ENDFOR

\STATE Sort the letters in alphabetical order

\FOR{each letter in sorted $freq$}
    \STATE Output letter and its frequency
\ENDFOR

\end{algorithmic}
\end{algorithm}
