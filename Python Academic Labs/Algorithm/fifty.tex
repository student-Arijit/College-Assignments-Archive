\begin{algorithm}
\caption{Read and Display the Last N Lines of a File}
\begin{algorithmic}[1]

\STATE Input file name:
\STATE $filename \gets$ user input

\STATE Input number of lines to read:
\STATE $n \gets$ user input

\STATE Open the file in read mode

\STATE Read all lines from the file:
\STATE $lines \gets$ all lines in the file

\STATE Select the last $n$ lines:
\STATE $last\_lines \gets lines[-n:]$

\FOR{each line in $last\_lines$}
    \STATE Display the line
\ENDFOR

\end{algorithmic}
\end{algorithm}
