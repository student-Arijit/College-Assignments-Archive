\begin{algorithm}
\caption{Demonstrate Subset and Superset Relations Between Sets}
\begin{algorithmic}[1]

\STATE Initialize Set 1:
\STATE $Set1 \gets \{1,2,3\}$

\STATE Initialize Set 2:
\STATE $Set2 \gets \{1,2,3,4,5\}$

\STATE Display Set 1 and Set 2

\STATE Check whether Set 1 is a subset of Set 2:
\STATE $isSubset \gets Set1.issubset(Set2)$

\STATE Display ``Set1 is subset of Set2:'' $isSubset$

\STATE Check whether Set 2 is a superset of Set 1:
\STATE $isSuperset \gets Set2.issuperset(Set1)$

\STATE Display ``Set2 is superset of Set1:'' $isSuperset$

\end{algorithmic}
\end{algorithm}
