\begin{algorithm}
\caption{Calculate Percentile Values of a NumPy Array}
\begin{algorithmic}[1]

\STATE Import NumPy library

\STATE Create array:
\STATE $arr \gets [10, 20, 30, 40, 50, 60, 70, 80]$

\STATE Compute percentiles:
\STATE $P_{25} \gets percentile(arr, 25)$
\STATE $P_{50} \gets percentile(arr, 50)$
\STATE $P_{75} \gets percentile(arr, 75)$

\STATE Display original array

\STATE Display 25th percentile

\STATE Display 50th percentile (median)

\STATE Display 75th percentile

\end{algorithmic}
\end{algorithm}
