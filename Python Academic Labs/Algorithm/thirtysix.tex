\begin{algorithm}
\caption{Unpack Elements of a Tuple}
\begin{algorithmic}[1]

\STATE Create a tuple:
\STATE $data \gets (10,\ 20,\ 30)$

\STATE Unpack tuple elements:
\STATE $a \gets 10$
\STATE $b \gets 20$
\STATE $c \gets 30$

\STATE Output ``Tuple: '' $data$

\STATE Output ``a = '' $a$

\STATE Output ``b = '' $b$

\STATE Output ``c = '' $c$

\end{algorithmic}
\end{algorithm}
