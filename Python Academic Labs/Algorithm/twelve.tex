\begin{algorithm}
\caption{Calculate Sum of Two Series}
\begin{algorithmic}[1]

\STATE Input value of $n$

\STATE $sumA \gets 0$

\FOR{$i \gets 1$ to $n$}
    \STATE $sumA \gets sumA + \dfrac{1}{i}$
\ENDFOR

\STATE $sumB \gets 0$

\FOR{$i \gets 1$ to $n$}
    \STATE $sumB \gets sumB + \dfrac{i}{i}$
\ENDFOR

\STATE Output ``Sum of series (a): '' $sumA$

\STATE Output ``Sum of series (b): '' $sumB$

\end{algorithmic}
\end{algorithm}
