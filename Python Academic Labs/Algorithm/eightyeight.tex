\begin{algorithm}
\caption{Interactive Altair Chart with Dropdown Filter}
\begin{algorithmic}[1]

\STATE Import Pandas and Altair libraries

\STATE Create dataset with columns Category, X, and Y

\STATE Define dropdown options:
\STATE $options \gets \{A, B, C\}$

\STATE Create dropdown selection bound to ``Category''

\STATE Create scatter plot chart with:
\STATE x-axis = X
\STATE y-axis = Y
\STATE color = Category

\STATE Attach dropdown selection to chart

\STATE Filter chart based on selected category

\STATE Set chart title: ``Interactive Altair Chart with Dropdown Filter''

\STATE Display interactive chart

\end{algorithmic}
\end{algorithm}
