\begin{algorithm}
\caption{Find the Size of a File}
\begin{algorithmic}[1]

\STATE Input file name:
\STATE $filename \gets$ user input

\STATE Create an object with the given file name

\STATE Obtain the size of the file:
\STATE $size \gets \text{getsize}(filename)$

\STATE Display the file size:
\STATE Output ``File size (bytes): '' $size$

\end{algorithmic}
\end{algorithm}
