\begin{algorithm}
\caption{Time-Series Line Chart with Tooltips using Altair}
\begin{algorithmic}[1]

\STATE Import Pandas, NumPy, and Altair libraries

\STATE Generate date range for 20 days starting from 2024-01-01

\STATE Generate random values for each date

\STATE Create DataFrame with columns:
\STATE $Date,\ Value$

\STATE Create line chart with points enabled

\STATE Map:
\STATE $x \rightarrow Date$
\STATE $y \rightarrow Value$

\STATE Add tooltips to display Date and Value on hover

\STATE Set chart title: ``Time-Series Line Chart with Tooltips''

\STATE Display interactive chart

\end{algorithmic}
\end{algorithm}
