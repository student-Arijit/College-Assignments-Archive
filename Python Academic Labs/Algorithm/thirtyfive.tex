\begin{algorithm}
\caption{Convert a List into a Tuple}
\begin{algorithmic}[1]

\STATE Create a list:
\STATE $my\_list \gets [10,\ 20,\ 30]$

\STATE Convert the list into a tuple:
\STATE $my\_tuple \gets tuple(my\_list)$

\STATE Output ``List: '' $my\_list$

\STATE Output ``Tuple: '' $my\_tuple$

\end{algorithmic}
\end{algorithm}
