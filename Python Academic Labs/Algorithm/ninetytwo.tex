\begin{algorithm}
\caption{Apply a Custom Function to a NumPy Array using Vectorization}
\begin{algorithmic}[1]

\STATE Import NumPy library

\STATE Define custom function:
\STATE $f(x) = x^2 + 2x + 1$

\STATE Create NumPy array:
\STATE $arr \gets [1, 2, 3, 4, 5]$

\STATE Vectorize the function:
\STATE $f_{vec} \gets vectorize(f)$

\STATE Apply vectorized function to array:
\STATE $result \gets f_{vec}(arr)$

\STATE Display original array

\STATE Display transformed array

\end{algorithmic}
\end{algorithm}
