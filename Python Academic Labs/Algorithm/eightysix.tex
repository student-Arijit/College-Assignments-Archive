\begin{algorithm}
\caption{Reshape a 1D NumPy Array into a 3D Array}
\begin{algorithmic}[1]

\STATE Import NumPy library

\STATE Create a 1D array with 24 elements:
\STATE $arr \gets [0, 1, 2, \dots, 23]$

\STATE Reshape the array into a 3D structure:
\STATE $arr_{3D} \gets reshape(arr,\ 2 \times 3 \times 4)$

\STATE Display original 1D array

\STATE Display reshaped 3D array

\end{algorithmic}
\end{algorithm}
