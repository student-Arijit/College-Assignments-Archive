\begin{algorithm}
\caption{Store Air Quality Index Records}
\begin{algorithmic}[1]

\STATE Create empty dictionary:
\STATE $aqi\_data \gets \{\}$

\STATE Input number of records:
\STATE $n \gets$ user input

\FOR{$i \gets 1$ to $n$}

    \STATE Input date:
    \STATE $date \gets$ user input

    \STATE Input AQI value:
    \STATE $aqi \gets$ user input

    \STATE $aqi\_data[date] \gets aqi$

\ENDFOR

\STATE Output ``Air Quality Dictionary:''

\STATE Output $aqi\_data$

\end{algorithmic}
\end{algorithm}
