\begin{algorithm}
\caption{Unzip a List of Tuples into Two Lists}
\begin{algorithmic}[1]

\STATE Create a list of tuples:
\STATE $data \gets [(1,A), (2,B), (3,C), (4,D)]$

\STATE $list1 \gets$ empty list

\STATE $list2 \gets$ empty list

\FOR{each tuple $(a,b)$ in $data$}
    \STATE Add $a$ to $list1$
    \STATE Add $b$ to $list2$
\ENDFOR

\STATE Output ``Original List of Tuples: '' $data$

\STATE Output ``List 1: '' $list1$

\STATE Output ``List 2: '' $list2$

\end{algorithmic}
\end{algorithm}
