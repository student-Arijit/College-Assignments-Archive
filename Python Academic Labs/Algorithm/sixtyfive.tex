\begin{algorithm}
\caption{Library Fine Calculation for Late Return of Books}
\begin{algorithmic}[1]

\STATE Create a Library record with:
\STATE $acc\_number, title, author, publisher$

\STATE Input book details:
\STATE Read accession number
\STATE Read title
\STATE Read author
\STATE Read publisher

\STATE Input number of late days:
\STATE $days\_late \gets$ user input

\STATE Compute fine:
\STATE $fine \gets days\_late \times 1.50$

\STATE Output fine:
\STATE ``Fine charged: \$'' $fine$

\end{algorithmic}
\end{algorithm}
