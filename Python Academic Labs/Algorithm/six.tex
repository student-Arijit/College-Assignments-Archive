\begin{algorithm}
\caption{Convert Celsius to Fahrenheit}
\begin{algorithmic}[1]

\STATE Input temperature in Celsius:
\STATE $celsius \gets$ user input

\STATE Compute Fahrenheit temperature:
\STATE $fahrenheit \gets \left(celsius \times \frac{9}{5}\right) + 32$

\STATE Output ``Temperature in Fahrenheit: '' $fahrenheit$ `` $^\circ$F''

\end{algorithmic}
\end{algorithm}
