\begin{algorithm}
\caption{Calculate Total Seconds in a Day}
\begin{algorithmic}[1]

\STATE Set number of hours in a day:
\STATE $hours \gets 24$

\STATE Set number of minutes in an hour:
\STATE $minutes \gets 60$

\STATE Set number of seconds in a minute:
\STATE $seconds \gets 60$

\STATE Compute total seconds in a day:
\STATE $total\_seconds \gets hours \times minutes \times seconds$

\STATE Output ``Total Seconds in a Day: '' $total\_seconds$

\end{algorithmic}
\end{algorithm}
