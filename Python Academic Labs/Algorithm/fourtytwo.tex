\begin{algorithm}
\caption{Find the Length of a Set}
\begin{algorithmic}[1]

\STATE Initialize a set:
\STATE $my\_set \gets \{10, 20, 30, 40, 50, 60\}$

\STATE Display the set:
\STATE Output ``Set: '' $my\_set$

\STATE Compute the length of the set:
\STATE $length \gets len(my\_set)$

\STATE Display the length:
\STATE Output ``Length of set: '' $length$

\end{algorithmic}
\end{algorithm}
