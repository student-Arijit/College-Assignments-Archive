\begin{algorithm}
\caption{Copy Contents of One CSV File to Another CSV File}
\begin{algorithmic}[1]

\STATE Input source CSV file name:
\STATE $source\_file \gets$ user input

\STATE Input destination CSV file name:
\STATE $destination\_file \gets$ user input

\STATE Open the source CSV file in read mode

\STATE Read all rows from the source file:
\STATE $rows \gets$ contents of the source CSV file

\STATE Open the destination CSV file in write mode

\STATE Write all rows to the destination file:
\STATE Write $rows$ into $destination\_file$

\STATE Display message:
\STATE Output ``CSV file copied successfully.''

\end{algorithmic}
\end{algorithm}
