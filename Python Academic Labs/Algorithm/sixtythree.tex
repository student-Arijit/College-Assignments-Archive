\begin{algorithm}
\caption{Find Roots of a Quadratic Equation}
\begin{algorithmic}[1]

\STATE Input coefficients:
\STATE $a \gets$ user input
\STATE $b \gets$ user input
\STATE $c \gets$ user input

\STATE Compute discriminant:
\STATE $d \gets b^2 - 4ac$

\STATE Compute first root:
\STATE $root1 \gets \frac{-b + \sqrt{d}}{2a}$

\STATE Compute second root:
\STATE $root2 \gets \frac{-b - \sqrt{d}}{2a}$

\STATE Output root1 and root2

\end{algorithmic}
\end{algorithm}
