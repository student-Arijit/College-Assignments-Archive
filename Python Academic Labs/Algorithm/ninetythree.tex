\caption{Convert Categorical Column into One-Hot Encoding using Pandas}
\begin{algorithmic}[1]

\STATE Create DataFrame with columns Name and City

\STATE Apply one-hot encoding on column ``City'':
\STATE $df_{encoded} \gets get\_dummies(df, columns = City)$

\STATE Display original DataFrame

\STATE Display one-hot encoded DataFrame
