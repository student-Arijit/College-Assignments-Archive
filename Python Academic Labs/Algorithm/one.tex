\begin{algorithm}
\caption{Average Calculation Program}
\begin{algorithmic}[1]

\STATE Start
\STATE Declare $m1, m2, m3$ as real numbers
\STATE Declare $sum, avg$ as real numbers
\STATE Read $m1, m2, m3$
\STATE $sum \leftarrow m1 + m2 + m3$
\STATE $avg \leftarrow sum / 3$
\STATE Display $avg$
\STATE Stop

\end{algorithmic}
\end{algorithm}
