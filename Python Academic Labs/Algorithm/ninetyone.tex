\begin{algorithm}
\caption{Heatmap Visualization using Altair}
\begin{algorithmic}[1]

\STATE Import Pandas, NumPy, and Altair libraries

\STATE Generate sample dataset with:
\STATE Categories $X \in \{A, B, C, D\}$
\STATE Categories $Y \in \{W, X, Y, Z\}$
\STATE Random numeric values

\STATE Create DataFrame with columns $X$, $Y$, and $Value$

\STATE Create heatmap using rectangular marks

\STATE Map:
\STATE $X \rightarrow$ x-axis
\STATE $Y \rightarrow$ y-axis
\STATE $Value \rightarrow$ color intensity (viridis scale)

\STATE Set chart title: ``Altair Heatmap Visualization''

\STATE Display heatmap chart

\end{algorithmic}
\end{algorithm}
