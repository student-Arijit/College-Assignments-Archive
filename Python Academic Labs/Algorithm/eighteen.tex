\begin{algorithm}
\caption{Concatenate Two Strings}
\begin{algorithmic}[1]

\STATE Input first string:
\STATE $str1 \gets$ user input

\STATE Input second string:
\STATE $str2 \gets$ user input

\STATE $result \gets$ empty string

\FOR{each character $ch$ in $str1$}
    \STATE Append $ch$ to $result$
\ENDFOR

\FOR{each character $ch$ in $str2$}
    \STATE Append $ch$ to $result$
\ENDFOR

\STATE Output ``Concatenated string: '' $result$

\end{algorithmic}
\end{algorithm}
