\begin{algorithm}
\caption{Arithmetic Operations on Two Pandas Series}
\begin{algorithmic}[1]

\STATE Import Pandas library

\STATE Create two Pandas Series:
\STATE $s_1 \gets [10, 20, 30, 40]$
\STATE $s_2 \gets [2, 4, 6, 8]$

\STATE Perform element-wise addition:
\STATE $add\_result \gets s_1 + s_2$

\STATE Perform element-wise subtraction:
\STATE $sub\_result \gets s_1 - s_2$

\STATE Perform element-wise multiplication:
\STATE $mul\_result \gets s_1 \times s_2$

\STATE Perform element-wise division:
\STATE $div\_result \gets s_1 / s_2$

\STATE Display both Series and results:
\STATE Output $s_1$, $s_2$
\STATE Output $add\_result$
\STATE Output $sub\_result$
\STATE Output $mul\_result$
\STATE Output $div\_result$

\end{algorithmic}
\end{algorithm}
