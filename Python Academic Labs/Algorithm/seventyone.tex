\begin{algorithm}
\caption{NumPy Array Creation Functions and Their Outputs}
\begin{algorithmic}[1]

\STATE Import NumPy library

\STATE Create array from list:
\STATE $arr_1 \gets array([1,2,3,4,5])$

\STATE Create zero matrix:
\STATE $arr_2 \gets zeros(3 \times 3)$

\STATE Create ones matrix:
\STATE $arr_3 \gets ones(2 \times 4)$

\STATE Create constant-filled array:
\STATE $arr_4 \gets full(2 \times 3,\ 7)$

\STATE Create range array:
\STATE $arr_5 \gets arange(1, 10, 2)$

\STATE Create evenly spaced values:
\STATE $arr_6 \gets linspace(0, 1, 5)$

\STATE Create identity matrix:
\STATE $arr_7 \gets eye(4)$

\STATE Create random matrix:
\STATE $arr_8 \gets random(2 \times 3)$

\STATE Display all arrays

\end{algorithmic}
\end{algorithm}
