\begin{algorithm}
\caption{Perform Basic Set Operations}
\begin{algorithmic}[1]

\STATE Create first set:
\STATE $set1 \gets \{1,2,3,4,5\}$

\STATE Create second set:
\STATE $set2 \gets \{4,5,6,7,8\}$

\STATE Compute intersection:
\STATE $intersection \gets set1 \cap set2$

\STATE Compute union:
\STATE $union \gets set1 \cup set2$

\STATE Compute set difference:
\STATE $difference \gets set1 - set2$

\STATE Compute symmetric difference:
\STATE $sym\_difference \gets set1 \triangle set2$

\STATE Output ``Set 1: '' $set1$

\STATE Output ``Set 2: '' $set2$

\STATE Output ``Intersection: '' $intersection$

\STATE Output ``Union: '' $union$

\STATE Output ``Set Difference (Set1 - Set2): '' $difference$

\STATE Output ``Symmetric Difference: '' $sym\_difference$

\end{algorithmic}
\end{algorithm}
