\begin{algorithm}
\caption{Compute Covariance Matrix of a Dataset using NumPy}
\begin{algorithmic}[1]

\STATE Import NumPy library

\STATE Define dataset where each row represents a variable and each column an observation:
\STATE $data \gets \begin{bmatrix}
2 & 4 & 6 & 8 & 10 \\
1 & 3 & 5 & 7 & 9 \\
10 & 20 & 30 & 40 & 50
\end{bmatrix}$

\STATE Compute covariance matrix:
\STATE $cov\_matrix \gets cov(data)$

\STATE Display dataset

\STATE Display covariance matrix

\end{algorithmic}
\end{algorithm}
