\begin{verbatim}
# Create a rolling window standard deviation column in Pandas

import pandas as pd

# Sample DataFrame
data = {
    "Values": [10, 12, 15, 13, 18, 20, 22, 25, 30, 28]
}

df = pd.DataFrame(data)

# Rolling window standard deviation (window size = 3)
df["Rolling_STD"] = df["Values"].rolling(window=3).std()

# Display result
print("DataFrame with Rolling Standard Deviation:")
print(df)
\end{verbatim}
