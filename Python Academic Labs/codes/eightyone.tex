\begin{verbatim}
import numpy as np
data = np.array([
    [2, 4, 6, 8, 10],
    [1, 3, 5, 7, 9],
    [10, 20, 30, 40, 50]
])

# Compute covariance matrix
cov_matrix = np.cov(data)

# Display result
print("Dataset:")
print(data)

print("Covariance Matrix:")
print(cov_matrix)
