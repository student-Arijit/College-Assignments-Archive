class AlphabetList:
    def __init__(self):
        self.letters = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h']

    def display_slices(self):
        print("First three letters:", self.letters[:3])
        print("Three letters from the middle:", self.letters[2:5])
        print("Letters from index 4 to end:", self.letters[4:])

obj = AlphabetList()
obj.display_slices()
