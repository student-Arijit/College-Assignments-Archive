class SetRelation:
    def __init__(self):
        self.set1 = {1, 2, 3}
        self.set2 = {1, 2, 3, 4, 5}

    def demonstrate(self):
        print("Set 1:", self.set1)
        print("Set 2:", self.set2)

        print("Set1 is subset of Set2:", self.set1.
        issubset(self.set2))
        print("Set2 is superset of Set1:", self.set2.
        issuperset(self.set1))

obj = SetRelation()
obj.demonstrate()
\end{verbatim}
