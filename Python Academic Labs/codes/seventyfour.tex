\begin{verbatim}
import pandas as pd
# Create two Pandas Series
s1 = pd.Series([10, 20, 30, 40])
s2 = pd.Series([2, 4, 6, 8])
# Addition
add_result = s1 + s2
# Subtraction
sub_result = s1 - s2
# Multiplication
mul_result = s1 * s2
# Division
div_result = s1 / s2
# Display results
print("Series 1:\n", s1)
print("Series 2:\n", s2)
print("\nAddition:\n", add_result)
print("\nSubtraction:\n", sub_result)
print("\nMultiplication:\n", mul_result)
print("\nDivision:\n", div_result)
\end{verbatim}
