\begin{verbatim}
import numpy as np
# 1. array() -> Create array from list/tuple
arr1 = np.array([1, 2, 3, 4, 5])
print("array():", arr1)
# 2. zeros() -> Create array filled with 0
arr2 = np.zeros((3, 3))
print("zeros():\n", arr2)
# 3. ones() -> Create array filled with 1
arr3 = np.ones((2, 4))
print("ones():\n", arr3)
# 4. full() -> Create array filled with a specific value
arr4 = np.full((2, 3), 7)
print("full():\n", arr4)
# 5. arange() -> Create array with range of values
arr5 = np.arange(1, 10, 2)
print("arange():", arr5)
# 6. linspace() -> Create evenly spaced values
arr6 = np.linspace(0, 1, 5)
print("linspace():", arr6)
# 7. eye() -> Identity matrix
arr7 = np.eye(4)
print("eye():\n", arr7)
# 8. random() -> Random values array
arr8 = np.random.rand(2, 3)
print("random():\n", arr8)
\end{verbatim}
