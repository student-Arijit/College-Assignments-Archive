\begin{verbatim}
import pandas as pd

# Create DataFrame
data = {
    "ID": [1, 2, 3, 4, 5, 6],
    "Name": ["Amit", "Rahul", "Sita", "Rahul", "John", "Sita"],
    "Marks": [80, 85, 90, 85, 88, 90]
}

df = pd.DataFrame(data)

# Remove duplicate rows based on "Name" column
df_unique = df.drop_duplicates(subset="Name")

# Display original DataFrame
print("Original DataFrame:")
print(df)

# Display DataFrame after removing duplicates
print("\nDataFrame after removing duplicates (based on Name):")
print(df_unique)
\end{verbatim}
