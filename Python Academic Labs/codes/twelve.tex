\begin{verbatim}
    import math

    class SeriesSum:
            def __init__(self, n):
                self.n = n
        
            def series_a(self):
                total = 0
                for i in range(1, self.n + 1):
                    total += 1 / i
                return total
        
            def series_b(self):
                total = 0
                for i in range(1, self.n + 1):
                    total += math.pow(i, i) / i
                return total
        
    n = int(input("Enter n: "))
    obj = SeriesSum(n)
    print("Sum of series (a):", obj.series_a())
    print("Sum of series (b):", obj.series_b())
\end{verbatim}
