# Compute pairwise correlation matrix of a DataFrame using Pandas

import pandas as pd

# Sample DataFrame
data = {
    "Math": [80, 90, 70, 85, 95],
    "Physics": [75, 88, 65, 80, 92],
    "Chemistry": [78, 85, 68, 82, 90],
    "English": [70, 75, 72, 78, 80]
}

df = pd.DataFrame(data)

# Compute correlation matrix
corr_matrix = df.corr()

# Display result
print("DataFrame:")
print(df)

print("\nCorrelation Matrix:")
print(corr_matrix)
