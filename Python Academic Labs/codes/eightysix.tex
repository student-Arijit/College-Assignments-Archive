\begin{verbatim}
# Reshape a 1D NumPy array into a 3D array

import numpy as np

# Create a 1D array with 24 elements
arr = np.arange(24)

# Reshape into a 3D array (2 × 3 × 4)
arr_3d = arr.reshape(2, 3, 4)

# Display results
print("Original 1D Array:")
print(arr)

print("Reshaped 3D Array (2×3×4):")
print(arr_3d)
\end{verbatim}
